\section{L3-L4}
\begin{enumerate}

	\item 
	since
	\begin{equation}
	\mathbf{r}=\mathbf{f}(\mathbf{c},t)=\mathbf{g}(\mathbf{c})h(t)
	\end{equation}
	$\Rightarrow$
	\begin{align}
	\mathbf{g}(\mathbf{c})=\frac{\mathbf{r}}{ h(t)}
	\end{align}
	$\Rightarrow$
	\begin{align}
		\mathbf{c}=\mathbf{g}^{-1}(\frac{\mathbf{r}}{ h(t)})
	\end{align}
	so
	\begin{align}
	\mathbf{v}&=\frac{\partial{\mathbf{r}}}{\partial t}\\
				&=\mathbf{g}(\mathbf{c})\dot{h}(t)\\
				&=\mathbf{g}(\mathbf{\mathbf{g}^{-1}(\frac{\mathbf{r}}{ h(t)})})\dot{h}(t)
	\end{align}
	
	\item 
	\begin{align}
	g(c)=c\\
	h(t)=t^2
	\end{align}
	and it is easy to find
	\begin{equation}
	g^{-1}(c)=c
	\end{equation}
	\begin{align}
	v&=g(g^{-1}(\frac{x}{h(t)}))\dot{h}(t)\\
	&=g^{-1}(\frac{x}{h(t)})\dot{h}(t)\\
		&=\frac{x}{h(t)}\dot{h}(t)\\
		&=2t\frac{x}{t^2}\\
		&=2\frac{x}{t}\\
	\end{align}
	
	
	\item
	at time $t$ the $c$th element is at $f(c,t)$ ,
	so the temperature of the $c$th element is
	\begin{align}
	T=g(f(c,t),t)
	\end{align}
	so the variation rate is
	\begin{align}
	\frac{dT}{dt}&=\frac{\partial g}{\partial x}\frac{df}{dt}+\frac{\partial g}{\partial t}
	\end{align}
	
	\item
	\begin{align}
	\left \{
	\begin{aligned}

	\right .
	\end{align}
	$\Rightarrow$	
		\begin{align}
		\left \{
		\begin{aligned}
			x=c_x(1+t)\\
			y=c_y(2+t)^2

		\right .
		\end{align}
	with the boundary condition
	\begin{align}
	\mathbf{x}(0)=\mathbf{x}_0
	\end{align}
	the path line through $\mathbf{x}_0$ is	
		\begin{align}
		\left \{
		\begin{aligned}
		x=x_0(1+t)\\
		y=\frac{y_0}{4}(2+t)^2
		\end{aligned}
		\right .
		\end{align}
		
		and the streamlines at $t=0$ is
	\begin{align}
	\left \{
	\begin{aligned}
	x=c_x\\
	y=2c_y
	\end{aligned}
	\right .
	\end{align}
		$\Rightarrow$
		\begin{eqnarray}
		y=2\frac{c_y}{c_x}x
		\end{eqnarray}
		
		\item
		\begin{align}
		\frac{\partial \rho}{\partial t}+\frac{\partial (\rho u)}{\partial x}=0
		\end{align}
		since
		\begin{equation}
		\rho=\rho_0(2-\cos\omega t)
		\end{equation}
		$\Rightarrow$
		\begin{align}
		\frac{\partial u}{\partial x}&=-\frac{\partial \rho}{\rho\partial t }\\
		&=\frac{\omega \sin \omega t }{\cos\omega t -2 }\\
		&:=f(t)

		u&=\int f(t) dx\\

		\Rightarrow&\frac{\partial (\rho h \delta x)}{\partial t}+\frac{\partial(\rho u h)}{\partial x}\delta x=0\\
		\Rightarrow&\frac{\partial  h}{\partial t}+\frac{\partial( u h)}{\partial x}=0

		momentum conservation
			\begin{align}
			&\frac{\partial (\rho u \delta v)}{\partial t}+\frac{\partial(\rho u^2 h+p)}{\partial x}\delta x=0\\
